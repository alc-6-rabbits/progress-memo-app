```javascript
// composables/useGoogleChatMd.js
export const useGoogleChatMd = () => {
  // 標準Markdown → Google Chat Markdown 変換
  const convertToGoogleChat = (standardMd) => {
    return standardMd
      .replace(/\*\*(.+?)\*\*/g, '*$1*')     // **bold** → *bold*
      .replace(/^- /gm, '・')                 // - list → ・list
      .replace(/^#{1,6}\s+/gm, '')            // # heading → heading
      .replace(/^> /gm, '＞ ')                // > quote → ＞ quote
      .replace(/\[([^\]]+)\]\([^\)]+\)/g, '$1') // [text](url) → text
  }

  // Google Chat 互換プレビュー用 HTML レンダリング
  const renderGoogleChatPreview = (text) => { ... }

  return { convertToGoogleChat, renderGoogleChatPreview }
}
```

**責務**: Google Chat の独自マークダウン仕様への変換処理。

#### バー位置の算出ロジック

```javascript
const barStyle = (item) => {
  if (!item.startDate || !item.dueDate) return { display: 'none' }
  
  const totalDays = dateRange.length
  const start = daysBetween(viewStartDate, new Date(item.startDate))
  const duration = daysBetween(new Date(item.startDate), new Date(item.dueDate)) + 1
  
  const leftPercent = (Math.max(0, start) / totalDays) * 100
  const widthPercent = (Math.min(duration, totalDays - start) / totalDays) * 100
  
  return {
    left: `${leftPercent}%`,
    width: `${Math.max(widthPercent, 1)}%`  // 最小1%幅を保証
  }
}

const barClass = (item) => ({
  'gantt-bar--overdue': item.scheduleStatus === 'overdue',
  'gantt-bar--at-risk': item.scheduleStatus === 'at-risk',
  'gantt-bar--on-track': item.scheduleStatus === 'on-track',
  'gantt-bar--completed': item.scheduleStatus === 'completed',
})
```

## 12. 技術的判断の記録

| 判断項目 | 選択 | 理由 |
|---------|------|------|
| ガントチャート実装方式 | CSS Grid | ライブラリ不要、Ace Theme との一貫性確保、必要機能が十分カバー可能 |
| プロジェクト階層化方式 | タグ方式 | 最小変更、柔軟性（複数タグ可）、タスクfrontmatter変更不要 |
| GitHub API 方式 | GraphQL (Projects V2) + REST (Events, Comments) | Projects V2 は GraphQL 必須、Events/Comments は REST が簡潔 |
| コンテキストメニュー | カスタム実装 (Vue) | Electron の native メニューではなく、Ace Theme に統一した Vue コンポーネント |
| 状態管理 | Composables (Vue 3 Composition API) | Vuex/Pinia 不要、composables で十分。既存アプリに状態管理ライブラリなし |
| Google Chat MD 変換 | 送信時変換 | エディタ内は標準MDで書き、送信のみ変換。プレビューで事前確認可能 |
